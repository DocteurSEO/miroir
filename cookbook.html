🪞 Welcome to Miroir.js!

Miroir.js is an ultra-lightweight (2KB) reactive data binding library with zero dependencies. In this comprehensive tutorial, you'll master:

  • Step 1: HTML Templates - Simple {{ variable }} syntax with automatic DOM updates
  • Step 2: JavaScript Bindings - CSS selector based configuration for precise control
  • Step 3: Extensions System - Create powerful custom directives
  • Step 4: Advanced Patterns - Real-world applications and best practices
  • Step 5: Form Validation - Build robust user input handling
  • Step 6: Dashboard Example - Real-time data updates and visualizations

Why Miroir.js? Perfect for adding reactivity to existing HTML, prototypes, widgets, and learning reactive programming concepts without framework overhead.

📊 Step 1: Simple Counter with HTML Templates

The easiest way to use Miroir.js is with HTML templates. Use {{ variable }} to display reactive data and d-model for two-way binding.

Key concepts: reactive state, template variables, automatic DOM updates

💻 Code
<!DOCTYPE html> <html> <head> <title>Counter App</title> </head> <body> <div class="demo-app"> <h2>Simple Counter</h2> <!-- HTML Template with {{ }} --> <div class="bind"> <div class="counter">{{ count }}</div> <p>You clicked {{ count }} times</p> </div> <button onclick="app.count++">+1</button> <button onclick="app.count--">-1</button> <button onclick="app.count = 0">Reset</button> </div> <script src="miroir.js"></script> <script> // Create reactive state const app = miroir.create({ count: 0 }); </script> </body> </html>

🚀 Step 3: Todo App with Extensions

Extensions are where Miroir.js shines! Create custom directives like m-show, m-each, m-count to handle complex scenarios with elegant simplicity.

Key concepts: custom directives, list rendering, conditional display, array manipulation

💻 Code
<!DOCTYPE html> <html> <body> <div class="demo-app"> <h2>Todo App</h2> <p>Tasks: <span class="task-count">0</span></p> <input id="new-task" placeholder="Add new task..." onkeypress="if(event.key==='Enter') addTask()"> <button onclick="addTask()">Add</button> <div id="todo-list"></div> <div class="empty-state" m-show="isEmpty"> No tasks yet! Add one above. </div> </div> <script src="miroir.js"></script> <script> // Load extensions miroir.extend('m-show', (el, prop, state) => { el.style.display = state[prop] ? 'block' : 'none'; }); miroir.extend('m-count', (el, prop, state) => { const value = state[prop]; el.textContent = Array.isArray(value) ? value.length : value; }); const app = miroir.create({ newTask: '', todos: [], isEmpty: true }, { bindings: [ { '#new-task': 'newTask' }, { '.task-count': 'todos' }, ] }); function addTask() { if (app.newTask.trim()) { app.todos.push({ id: Date.now(), text: app.newTask.trim(), done: false }); app.newTask = ''; app.isEmpty = app.todos.length === 0; renderTodos(); } } function renderTodos() { const list = document.getElementById('todo-list'); list.innerHTML = app.todos.map((todo, i) => ` <div class="todo-item ${todo.done ? 'completed' : ''}"> <input type="checkbox" ${todo.done ? 'checked' : ''} onchange="toggleTodo(${i})"> <span>${todo.text}</span> <button onclick="deleteTodo(${i})" style="margin-left: auto;">×</button> </div> `).join(''); } function toggleTodo(index) { app.todos[index].done = !app.todos[index].done; renderTodos(); } function deleteTodo(index) { app.todos.splice(index, 1); app.isEmpty = app.todos.length === 0; renderTodos(); } </script> </body> </html>

🎆 Step 4: Advanced Extensions & Patterns

Explore powerful extensions like m-class, m-each, and m-validate. Learn how to create dynamic lists, conditional styling, and input validation.

Key concepts: advanced directives, dynamic classes, list iteration, validation patterns

💻 Code
<!DOCTYPE html> <html> <body> <div class="demo-app"> <h2>Advanced Extensions Demo</h2> <!-- Dynamic Classes --> <div class="status-indicator" m-class="active:isOnline"> Status: <span class="status-text">{{ status }}</span> </div> <button onclick="app.isOnline = !app.isOnline"> Toggle Status </button> <!-- Shopping List with m-each --> <h3>Shopping List (<span m-count="items">0</span> items)</h3> <input d-model="newItem" placeholder="Add item..." onkeypress="if(event.key==='Enter') addItem()"> <button onclick="addItem()">Add Item</button> <div class="item-list" m-each="item in items"></div> <!-- Form Validation --> <h3>Email Validation</h3> <input d-model="email" placeholder="Enter email" m-validate="email" class="form-input"> <div class="validation-message" m-show="emailError"> Please enter a valid email address </div> </div> <script src="miroir.js"></script> <script> // Define advanced extensions miroir.extend('m-class', (el, expr, state) => { const [className, prop] = expr.split(':'); el.classList.toggle(className, !!state[prop]); }); miroir.extend('m-each', (el, expr, state) => { const [itemName, arrayProp] = expr.split(' in '); const array = state[arrayProp] || []; el.innerHTML = array.map((item, i) => ` <div class="list-item"> <span>${item}</span> <button onclick="removeItem(${i})">×</button> </div> `).join(''); }); miroir.extend('m-count', (el, prop, state) => { const value = state[prop]; el.textContent = Array.isArray(value) ? value.length : value; }); miroir.extend('m-validate', (el, rules, state) => { const prop = el.getAttribute('d-model'); const value = state[prop] || ''; const isValid = rules === 'email' ? /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value) : true; el.classList.toggle('error', !isValid && value.length > 0); state[prop + 'Error'] = !isValid && value.length > 0; }); const app = miroir.create({ isOnline: false, status: 'Offline', newItem: '', items: ['Milk', 'Bread', 'Eggs'], email: '', emailError: false }); // Watch status changes miroir.watch('isOnline', (online) => { app.status = online ? 'Online' : 'Offline'; }); function addItem() { if (app.newItem.trim()) { app.items.push(app.newItem.trim()); app.newItem = ''; } } function removeItem(index) { app.items.splice(index, 1); } </script> </body> </html>

📋 Step 5: Advanced Form Validation

Build robust forms with real-time validation, custom rules, and user-friendly error messages using Miroir.js extensions.

Key concepts: validation rules, error handling, user experience patterns

💻 Code
<!DOCTYPE html> <html> <body> <div class="demo-app"> <h2>Registration Form</h2> <form class="validation-form"> <div class="form-group"> <label>Username:</label> <input d-model="username" m-validate="required|minLength:3" placeholder="At least 3 characters"> <div class="error-message" m-show="usernameError">{{ usernameErrorMsg }}</div> </div> <div class="form-group"> <label>Email:</label> <input d-model="email" m-validate="required|email" placeholder="your@email.com"> <div class="error-message" m-show="emailError">{{ emailErrorMsg }}</div> </div> <div class="form-group"> <label>Password:</label> <input type="password" d-model="password" m-validate="required|minLength:6" placeholder="At least 6 characters"> <div class="error-message" m-show="passwordError">{{ passwordErrorMsg }}</div> </div> <div class="form-group"> <label>Confirm Password:</label> <input type="password" d-model="confirmPassword" m-validate="required|match:password" placeholder="Repeat password"> <div class="error-message" m-show="confirmPasswordError">{{ confirmPasswordErrorMsg }}</div> </div> <button type="button" onclick="submitForm()" class="submit-btn" m-class="disabled:!formValid"> Register </button> <div class="success-message" m-show="submitted"> ✓ Registration successful! </div> </form> </div> <script src="miroir.js"></script> <script> // Enhanced validation extension miroir.extend('m-validate', (el, rules, state) => { const prop = el.getAttribute('d-model'); const value = state[prop] || ''; const ruleList = rules.split('|'); let errors = []; ruleList.forEach(rule => { if (rule === 'required' && !value.trim()) { errors.push('This field is required'); } else if (rule === 'email' && value && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value)) { errors.push('Please enter a valid email'); } else if (rule.startsWith('minLength:')) { const minLen = parseInt(rule.split(':')[1]); if (value && value.length < minLen) { errors.push(`Minimum ${minLen} characters required`); } } else if (rule.startsWith('match:')) { const matchProp = rule.split(':')[1]; if (value && value !== state[matchProp]) { errors.push('Passwords do not match'); } } }); const hasError = errors.length > 0; el.classList.toggle('error', hasError); state[prop + 'Error'] = hasError; state[prop + 'ErrorMsg'] = errors[0] || ''; }); const app = miroir.create({ username: '', email: '', password: '', confirmPassword: '', usernameError: false, emailError: false, passwordError: false, confirmPasswordError: false, usernameErrorMsg: '', emailErrorMsg: '', passwordErrorMsg: '', confirmPasswordErrorMsg: '', formValid: false, submitted: false }); // Watch all fields for form validity ['username', 'email', 'password', 'confirmPassword'].forEach(field => { miroir.watch(field, () => { updateFormValidity(); }); }); function updateFormValidity() { app.formValid = !app.usernameError && !app.emailError && !app.passwordError && !app.confirmPasswordError && app.username && app.email && app.password && app.confirmPassword; } function submitForm() { if (app.formValid) { app.submitted = true; setTimeout(() => app.submitted = false, 3000); } } </script> </body> </html>

📊 Step 6: Real-time Dashboard

Create a live dashboard with automatic updates, data visualization, and interactive controls using Miroir.js watchers and extensions.

Key concepts: real-time updates, data formatting, intervals, computed values

💻 Code
<!DOCTYPE html> <html> <body> <div class="demo-app"> <h2>Live Dashboard</h2> <div class="controls"> <button onclick="toggleUpdates()">{{ isUpdating ? 'Pause' : 'Start' }} Updates</button> <input d-model="updateInterval" type="number" placeholder="Interval (ms)" min="1000" max="10000"> </div> <div class="stats-grid"> <div class="stat-card"> <h3>Users Online</h3> <div class="stat-value" m-animate="users">{{ users }}</div> <div class="stat-change" m-class="positive:usersTrend>0|negative:usersTrend<0"> {{ usersTrend > 0 ? '+' : '' }}{{ usersTrend }} </div> </div> <div class="stat-card"> <h3>Revenue</h3> <div class="stat-value" m-currency="revenue">$0</div> <div class="stat-change" m-class="positive:revenueTrend>0|negative:revenueTrend<0"> {{ revenueTrend > 0 ? '+' : '' }}{{ revenueTrend | currency }} </div> </div> <div class="stat-card"> <h3>Server Load</h3> <div class="stat-value">{{ serverLoad }}%</div> <div class="progress-bar"> <div class="progress-fill" m-style="width:{{ serverLoad }}%"></div> </div> </div> </div> <div class="activity-log"> <h3>Recent Activity</h3> <div class="log-entries" m-each="entry in activityLog"></div> </div> </div> <script src="miroir.js"></script> <script> // Dashboard extensions miroir.extend('m-currency', (el, prop, state) => { const value = state[prop] || 0; el.textContent = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(value); }); miroir.extend('m-style', (el, expr, state) => { // Simple style binding: "property:value" const [property, value] = expr.split(':'); const processedValue = value.replace(/\{\{(.*?)\}\}/g, (match, prop) => { return state[prop.trim()] || 0; }); el.style[property] = processedValue; }); miroir.extend('m-animate', (el, prop, state) => { const newValue = state[prop]; const oldValue = parseInt(el.textContent) || 0; if (newValue !== oldValue) { el.style.transition = 'transform 0.3s ease'; el.style.transform = 'scale(1.1)'; setTimeout(() => { el.style.transform = 'scale(1)'; }, 300); } el.textContent = newValue; }); const dashboard = miroir.create({ users: 1250, revenue: 25000, serverLoad: 45, usersTrend: 0, revenueTrend: 0, isUpdating: false, updateInterval: 3000, activityLog: [ { time: '10:30', action: 'User joined', user: 'john_doe' }, { time: '10:28', action: 'Order placed', amount: '$125' }, { time: '10:25', action: 'Payment processed', id: '#1234' } ] }); let updateTimer; // Watch for interval changes miroir.watch('updateInterval', (interval) => { if (dashboard.isUpdating) { clearInterval(updateTimer); startUpdates(); } }); function toggleUpdates() { dashboard.isUpdating = !dashboard.isUpdating; if (dashboard.isUpdating) { startUpdates(); } else { clearInterval(updateTimer); } } function startUpdates() { updateTimer = setInterval(() => { // Simulate data changes const userChange = Math.floor(Math.random() * 20) - 10; const revenueChange = Math.floor(Math.random() * 2000) - 1000; const loadChange = Math.floor(Math.random() * 20) - 10; dashboard.usersTrend = userChange; dashboard.revenueTrend = revenueChange; dashboard.users = Math.max(0, dashboard.users + userChange); dashboard.revenue = Math.max(0, dashboard.revenue + revenueChange); dashboard.serverLoad = Math.max(0, Math.min(100, dashboard.serverLoad + loadChange)); // Add activity log entry const activities = [ 'User joined', 'Order placed', 'Payment processed', 'Server restarted', 'Backup completed' ]; const randomActivity = activities[Math.floor(Math.random() * activities.length)]; dashboard.activityLog.unshift({ time: new Date().toLocaleTimeString().slice(0, 5), action: randomActivity, user: 'user_' + Math.floor(Math.random() * 1000) }); // Keep only last 5 entries if (dashboard.activityLog.length > 5) { dashboard.activityLog = dashboard.activityLog.slice(0, 5); } }, dashboard.updateInterval || 3000); } </script> </body> </html>
